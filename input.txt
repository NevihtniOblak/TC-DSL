
procedure bubbleSort(array, len){

    for(var i = 0; len-1){
        for(var j = 0; len-i-2){
            if(array[j] > array[j+1]){
                var temp = array[j];
                array[j] = array[j+1];
                array[j+1] = temp;
            }
        }
    }
}
